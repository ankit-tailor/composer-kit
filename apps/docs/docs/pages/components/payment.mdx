import {

        tokenAddress="0x765de816845861e75a25fca122bb6898b8b1282a"

        <button
          onClick={() => {
            setIsOpen(!isOpen);
          }}
        >

        <PaymentDialog
          onOpenChange={() => {
            setIsOpen(!isOpen);
          }}
          open={isOpen}
        />
        <PaymentError />
      </Payment>
    </div>

}
